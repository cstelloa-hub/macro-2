Private Sub ProcesarFondo(datos As Variant, ByVal nFilas As Long, _
                          ByVal colIni As Long, ByVal colFin As Long, _
                          ByVal etiqueta As String, _
                          dicTipos As Object, dicMeses As Object, ordenFondos As Object)

    Dim r As Long
    r = FILA_ENCABEZADO_DIARIO + 1

    Do While r <= nFilas
        Dim etq As String
        etq = UCase(Trim(CStr(datos(r, colIni) & "")))

        If etq = "F1" Or etq = "F2" Or etq = "F3" Then
            Dim tipo As String: tipo = etq
            Dim filaHdr As Long: filaHdr = r

            ' Mapear columnas CC/DD/NAV de ESTE sub-bloque
            Dim colCC As Long, colDD As Long, colNAV As Long
            colCC = 0: colDD = 0: colNAV = 0
            Dim c As Long, h As String
            For c = colIni To colFin
                h = UCase(Trim(CStr(datos(filaHdr, c) & "")))
                Select Case h
                    Case "CC": colCC = c
                    Case "DD": colDD = c
                    Case "NAV": colNAV = c
                End Select
            Next c

            ' Obtener (o crear) dicFondos de este tipo y dicMes de este fondo
            Dim dicFondosTipo As Object
            If dicTipos.Exists(tipo) Then
                Set dicFondosTipo = dicTipos(tipo)
            Else
                Set dicFondosTipo = CreateObject("Scripting.Dictionary")
                dicTipos.Add tipo, dicFondosTipo
            End If

            Dim dicMes As Object
            If dicFondosTipo.Exists(etiqueta) Then
                Set dicMes = dicFondosTipo(etiqueta)
            Else
                Set dicMes = CreateObject("Scripting.Dictionary")
                dicFondosTipo.Add etiqueta, dicMes
                Dim claveOrden As String: claveOrden = tipo & "|" & etiqueta
                If Not ordenFondos.Exists(claveOrden) Then ordenFondos.Add claveOrden, ordenFondos.Count
            End If

            ' Leer las fechas de ESTE sub-bloque, fila por fila hacia abajo.
            ' El sub-bloque termina en cuanto aparece algo que NO es fecha válida
            ' (código SBS, "MONEDA?", otro F1/F2/F3, o se acaba la hoja),
            ' PERO toleramos celdas vacías intermedias antes de los datos.
            Dim rr As Long: rr = filaHdr + 1
            Dim yaLeyoDatos As Boolean: yaLeyoDatos = False

            Do While rr <= nFilas
                Dim valFecha As Variant: valFecha = datos(rr, colIni)
                Dim sCelda As String: sCelda = UCase(Trim(CStr(valFecha & "")))

                ' Otro encabezado de tipo -> fin de este sub-bloque
                If sCelda = "F1" Or sCelda = "F2" Or sCelda = "F3" Then Exit Do

                Dim clave As String: clave = ParsearMesAnio(valFecha)

                If clave <> "" Then
                    ' Es una fecha válida: leer datos
                    yaLeyoDatos = True
                    Dim cc As Double, dd As Double, nav As Double
                    cc = LeerNumero(datos, rr, colCC)
                    dd = LeerNumero(datos, rr, colDD)
                    nav = LeerNumero(datos, rr, colNAV)

                    If dicMes.Exists(clave) Then
                        Dim arr As Variant: arr = dicMes(clave)
                        arr(2) = nav            ' ajuste: último NAV manda
                        dicMes(clave) = arr
                    Else
                        dicMes.Add clave, Array(cc, dd, nav)
                    End If
                    If Not dicMeses.Exists(clave) Then dicMeses.Add clave, True
                Else
                    ' No es fecha válida
                    If sCelda = "" Then
                        ' Celda vacía: si todavía no empezaron los datos, tolerar y seguir.
                        ' Si ya leímos datos, una vacía marca el fin del sub-bloque.
                        If yaLeyoDatos Then Exit Do
                    Else
                        ' Texto/código/MONEDA? -> fin de este sub-bloque
                        Exit Do
                    End If
                End If

                rr = rr + 1
            Loop

            r = rr
        Else
            r = r + 1
        End If
    Loop
End Sub
